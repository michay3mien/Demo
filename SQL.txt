CREATE TABLE KHACHHANG
(
  MaKH INT NOT NULL,
  Ho VARCHAR(50) NOT NULL,
  GMAIL VARCHAR(50) NOT NULL,
  SDT INT NOT NULL,
  TenKH VARCHAR(50) NOT NULL,
  NgaySinh DATE NOT NULL,
  PRIMARY KEY (MaKH)
);

CREATE TABLE LOAIBANH
(
  MaLoaiBanh INT NOT NULL,
  TenLoaiBanh VARCHAR(50) NOT NULL,
  PRIMARY KEY (MaLoaiBanh)
);

CREATE TABLE BANH
(
  MaBanh INT NOT NULL,
  LoaiBanh INT NOT NULL,
  GiaTien INT NOT NULL,
  TenBanh VARCHAR(50) NOT NULL,
  SoLuong INT NOT NULL,
  MoTa VARCHAR(100) NOT NULL,
  MaLoaiBanh INT NOT NULL,
  PRIMARY KEY (MaBanh),
  FOREIGN KEY (MaLoaiBanh) REFERENCES LOAIBANH(MaLoaiBanh)
);

CREATE TABLE CHITIETHD
(
  MaCTHD INT NOT NULL,
  GiaTien INT NOT NULL,
  SoLuong INT NOT NULL,
  Loai INT NOT NULL,
  NgayMua DATE NOT NULL,
  GioMua VARCHAR(50) NOT NULL,
  TenBanh VARCHAR(50) NOT NULL,
  MaBanh INT NOT NULL,
  PRIMARY KEY (MaCTHD),
  FOREIGN KEY (MaBanh) REFERENCES BANH(MaBanh)
);

CREATE TABLE HOADON
(
  MaHD INT NOT NULL,
  SoLuong INT NOT NULL,
  GiaTien INT NOT NULL,
  TenBanh VARCHAR(50) NOT NULL,
  Loai VARCHAR(50) NOT NULL,
  MaCTHD INT NOT NULL,
  MaKH INT NOT NULL,
  PRIMARY KEY (MaHD),
  FOREIGN KEY (MaCTHD) REFERENCES CHITIETHD(MaCTHD),
  FOREIGN KEY (MaKH) REFERENCES KHACHHANG(MaKH)
);
